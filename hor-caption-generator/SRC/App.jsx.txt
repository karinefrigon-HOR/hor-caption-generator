import { useState, useEffect } from "react";

const PILLARS = ["Shadow Work", "Ritual Living", "Product Spotlight", "Community", "Education", "Seasonal / Event", "Testimonial"];
const PLATFORMS = ["Instagram", "Facebook", "Pinterest", "TikTok", "YouTube"];
const TONES = ["Mystical & Deep", "Warm & Nurturing", "Direct & Bold", "Poetic & Soft"];

const PLATFORM_ICONS = {
  Instagram: "📸", Facebook: "💬", Pinterest: "📌", TikTok: "🎵", YouTube: "🎥"
};

const PILLAR_ICONS = {
  "Shadow Work": "🖤", "Ritual Living": "🕯️", "Product Spotlight": "🛍️",
  "Community": "💫", "Education": "📚", "Seasonal / Event": "🌿", "Testimonial": "⭐"
};

const PLATFORM_TIPS = {
  Instagram: "Hook + line breaks + 3-5 hashtags",
  Facebook: "Conversational, question or poll ending",
  Pinterest: "SEO keywords first, describe the product/content clearly",
  TikTok: "Strong POV hook, casual, trending language",
  YouTube: "Searchable title style, what the viewer will learn"
};

export default function CaptionGenerator() {
  const [pillar, setPillar] = useState("");
  const [platform, setPlatform] = useState("");
  const [topic, setTopic] = useState("");
  const [tone, setTone] = useState("Mystical & Deep");
  const [count, setCount] = useState(3);
  const [loading, setLoading] = useState(false);
  const [captions, setCaptions] = useState([]);
  const [copied, setCopied] = useState(null);
  const [error, setError] = useState("");
  const [particles, setParticles] = useState([]);

  useEffect(() => {
    const p = Array.from({ length: 18 }, (_, i) => ({
      id: i,
      x: Math.random() * 100,
      y: Math.random() * 100,
      size: Math.random() * 3 + 1,
      delay: Math.random() * 6,
      duration: Math.random() * 8 + 6,
    }));
    setParticles(p);
  }, []);

  const generate = async () => {
    if (!pillar || !platform || !topic.trim()) {
      setError("Please fill in all fields before generating.");
      return;
    }
    setError("");
    setLoading(true);
    setCaptions([]);

    const systemPrompt = `You are a content strategist and copywriter for "Home of Rituals" — a mystical, emotionally deep digital brand selling shadow work journals, ritual tools, and self-discovery products, primarily to spiritual women aged 25-45. 

Brand voice: ${tone === "Mystical & Deep" ? "Dark, poetic, introspective. Uses moon/shadow/depth metaphors. Never toxic positivity." : tone === "Warm & Nurturing" ? "Soft, supportive, like a wise friend. Gentle but honest." : tone === "Direct & Bold" ? "No-fluff, powerful, cuts straight to the emotional truth." : "Lyrical, evocative, uses metaphor and imagery heavily."} 

Platform-specific rules:
- Instagram: Strong opening line, line breaks for breathing room, 1-2 relevant emojis max, end with a soft CTA or question, 3-5 hashtags
- Facebook: Conversational, warmer tone, longer if needed, end with question or poll prompt, no hashtags
- Pinterest: SEO-first description, keyword-rich, product-focused, clear value proposition, include "instant download" or "printable" if product-related
- TikTok: POV or hook format, casual and punchy, trending language, short, relatable
- YouTube: Search-optimized title or description style, clear "what you'll learn/get", friendly but professional

IMPORTANT: Return ONLY a valid JSON array, no markdown, no explanation. Format:
[{"caption": "full caption text here", "hook": "first line only", "hashtags": "#tag1 #tag2"}]`;

    const userPrompt = `Generate ${count} different ${platform} captions for Home of Rituals.
Content Pillar: ${pillar}
Topic: ${topic}
Tone: ${tone}

Make each caption distinctly different in approach (e.g., one story-based, one question-based, one bold statement). Return only the JSON array.`;

    try {
      const res = await fetch("https://api.anthropic.com/v1/messages", {
        method: "POST",
   headers: {
  "Content-Type": "application/json",
  "x-api-key": "YOUR_API_KEY_HERE",
  "anthropic-version": "2023-06-01"
},
```

⚠️ Replace `YOUR_API_KEY_HERE` with your actual Anthropic API key — the one that starts with `sk-ant-...`

---

Your folder should now look like:
```
hor-caption-generator/
├── index.html
├── package.json
├── vite.config.js
└── src/
    ├── main.jsx
    └── App.jsx
        body: JSON.stringify({
          model: "claude-sonnet-4-20250514",
          max_tokens: 1000,
          system: systemPrompt,
          messages: [{ role: "user", content: userPrompt }]
        })
      });
      const data = await res.json();
      const raw = data.content?.[0]?.text || "[]";
      const clean = raw.replace(/```json|```/g, "").trim();
      const parsed = JSON.parse(clean);
      setCaptions(parsed);
    } catch (e) {
      setError("Something went wrong. Please try again.");
    }
    setLoading(false);
  };

  const copyCaption = (text, idx) => {
    navigator.clipboard.writeText(text);
    setCopied(idx);
    setTimeout(() => setCopied(null), 2000);
  };

  return (
    <div style={{
      minHeight: "100vh",
      background: "#0d0b1a",
      fontFamily: "'Georgia', serif",
      color: "#f0ead6",
      position: "relative",
      overflow: "hidden",
    }}>
      <style>{`
        @import url('https://fonts.googleapis.com/css2?family=Playfair+Display:ital,wght@0,400;0,700;1,400&family=Lato:wght@300;400&display=swap');
        * { box-sizing: border-box; }
        .particle { position: absolute; border-radius: 50%; background: radial-gradient(circle, #c9a84c55, transparent); animation: float linear infinite; pointer-events: none; }
        @keyframes float { 0% { transform: translateY(100vh) rotate(0deg); opacity: 0; } 10% { opacity: 1; } 90% { opacity: 0.4; } 100% { transform: translateY(-10vh) rotate(360deg); opacity: 0; } }
        @keyframes shimmer { 0%,100% { opacity: 0.6; } 50% { opacity: 1; } }
        @keyframes fadeIn { from { opacity: 0; transform: translateY(16px); } to { opacity: 1; transform: translateY(0); } }
        @keyframes pulse { 0%,100% { box-shadow: 0 0 0 0 #c9a84c44; } 50% { box-shadow: 0 0 0 8px transparent; } }
        @keyframes spin { to { transform: rotate(360deg); } }
        .select-field, .text-field {
          width: 100%; background: #1a1730; border: 1px solid #3a2f5a; 
          color: #f0ead6; border-radius: 10px; padding: 12px 16px;
          font-family: 'Lato', sans-serif; font-size: 14px; outline: none;
          transition: border-color 0.3s, box-shadow 0.3s;
          appearance: none; cursor: pointer;
        }
        .select-field:focus, .text-field:focus { border-color: #c9a84c; box-shadow: 0 0 0 3px #c9a84c22; }
        .generate-btn {
          width: 100%; padding: 16px; background: linear-gradient(135deg, #c9a84c, #a07830);
          border: none; border-radius: 12px; color: #0d0b1a; font-family: 'Playfair Display', serif;
          font-size: 16px; font-weight: 700; cursor: pointer; transition: all 0.3s;
          animation: pulse 2.5s infinite; letter-spacing: 0.5px;
        }
        .generate-btn:hover { transform: translateY(-2px); box-shadow: 0 8px 24px #c9a84c55; }
        .generate-btn:disabled { opacity: 0.5; cursor: not-allowed; animation: none; transform: none; }
        .caption-card {
          background: #16132a; border: 1px solid #2e2650; border-radius: 14px;
          padding: 24px; animation: fadeIn 0.5s ease forwards; transition: border-color 0.3s;
        }
        .caption-card:hover { border-color: #c9a84c55; }
        .copy-btn {
          padding: 8px 18px; background: transparent; border: 1px solid #c9a84c;
          color: #c9a84c; border-radius: 8px; font-family: 'Lato', sans-serif;
          font-size: 12px; cursor: pointer; transition: all 0.2s; letter-spacing: 0.5px;
        }
        .copy-btn:hover, .copy-btn.copied { background: #c9a84c; color: #0d0b1a; }
        .pill {
          display: inline-block; padding: 4px 12px; background: #2a2145;
          border-radius: 20px; font-size: 11px; color: #c9a84c;
          font-family: 'Lato', sans-serif; letter-spacing: 0.5px; text-transform: uppercase;
        }
        .tip-box {
          background: #1e1a35; border-left: 3px solid #c9a84c; border-radius: 0 8px 8px 0;
          padding: 10px 14px; font-family: 'Lato', sans-serif; font-size: 12px; color: #a89060;
        }
        .counter-btn {
          width: 32px; height: 32px; border-radius: 8px; border: 1px solid #3a2f5a;
          background: #1a1730; color: #f0ead6; font-size: 18px; cursor: pointer;
          display: flex; align-items: center; justify-content: center; transition: all 0.2s;
        }
        .counter-btn:hover { border-color: #c9a84c; color: #c9a84c; }
        .spinner { width: 28px; height: 28px; border: 3px solid #3a2f5a; border-top-color: #c9a84c; border-radius: 50%; animation: spin 0.8s linear infinite; margin: 0 auto; }
      `}</style>

      {/* Particles */}
      {particles.map(p => (
        <div key={p.id} className="particle" style={{
          left: `${p.x}%`, width: p.size * 4, height: p.size * 4,
          animationDuration: `${p.duration}s`, animationDelay: `${p.delay}s`,
        }} />
      ))}

      {/* Glow orbs */}
      <div style={{ position: "fixed", top: "-20%", right: "-10%", width: 500, height: 500, borderRadius: "50%", background: "radial-gradient(circle, #2d1f5e44, transparent 70%)", pointerEvents: "none" }} />
      <div style={{ position: "fixed", bottom: "-20%", left: "-10%", width: 400, height: 400, borderRadius: "50%", background: "radial-gradient(circle, #1a3a2a33, transparent 70%)", pointerEvents: "none" }} />

      <div style={{ maxWidth: 680, margin: "0 auto", padding: "40px 20px", position: "relative", zIndex: 1 }}>

        {/* Header */}
        <div style={{ textAlign: "center", marginBottom: 40 }}>
          <div style={{ fontSize: 11, letterSpacing: 4, color: "#c9a84c", fontFamily: "'Lato', sans-serif", textTransform: "uppercase", marginBottom: 12, animation: "shimmer 3s infinite" }}>
            ✦ Home of Rituals ✦
          </div>
          <h1 style={{ fontFamily: "'Playfair Display', serif", fontSize: 36, fontWeight: 400, margin: "0 0 8px", lineHeight: 1.2 }}>
            Caption <em style={{ color: "#c9a84c" }}>Generator</em>
          </h1>
          <p style={{ fontFamily: "'Lato', sans-serif", fontSize: 14, color: "#7a6f8a", margin: 0, fontWeight: 300 }}>
            Craft platform-perfect captions in your brand voice
          </p>
        </div>

        {/* Form card */}
        <div style={{ background: "#13112299", backdropFilter: "blur(12px)", border: "1px solid #2e2650", borderRadius: 20, padding: 32, marginBottom: 32 }}>

          {/* Row 1: Pillar + Platform */}
          <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr", gap: 16, marginBottom: 20 }}>
            <div>
              <label style={{ display: "block", fontFamily: "'Lato', sans-serif", fontSize: 11, letterSpacing: 2, color: "#c9a84c", textTransform: "uppercase", marginBottom: 8 }}>
                Content Pillar
              </label>
              <select className="select-field" value={pillar} onChange={e => setPillar(e.target.value)}>
                <option value="">Select pillar...</option>
                {PILLARS.map(p => <option key={p} value={p}>{PILLAR_ICONS[p]} {p}</option>)}
              </select>
            </div>
            <div>
              <label style={{ display: "block", fontFamily: "'Lato', sans-serif", fontSize: 11, letterSpacing: 2, color: "#c9a84c", textTransform: "uppercase", marginBottom: 8 }}>
                Platform
              </label>
              <select className="select-field" value={platform} onChange={e => setPlatform(e.target.value)}>
                <option value="">Select platform...</option>
                {PLATFORMS.map(p => <option key={p} value={p}>{PLATFORM_ICONS[p]} {p}</option>)}
              </select>
            </div>
          </div>

          {/* Platform tip */}
          {platform && (
            <div className="tip-box" style={{ marginBottom: 20 }}>
              💡 <strong>{platform}:</strong> {PLATFORM_TIPS[platform]}
            </div>
          )}

          {/* Topic */}
          <div style={{ marginBottom: 20 }}>
            <label style={{ display: "block", fontFamily: "'Lato', sans-serif", fontSize: 11, letterSpacing: 2, color: "#c9a84c", textTransform: "uppercase", marginBottom: 8 }}>
              Topic / Theme
            </label>
            <textarea
              className="text-field"
              rows={3}
              placeholder="e.g. La Peur d'Être Vue — the fear of being seen and how shadow work helps..."
              value={topic}
              onChange={e => setTopic(e.target.value)}
              style={{ resize: "vertical" }}
            />
          </div>

          {/* Row 2: Tone + Count */}
          <div style={{ display: "grid", gridTemplateColumns: "1fr auto", gap: 16, marginBottom: 24, alignItems: "end" }}>
            <div>
              <label style={{ display: "block", fontFamily: "'Lato', sans-serif", fontSize: 11, letterSpacing: 2, color: "#c9a84c", textTransform: "uppercase", marginBottom: 8 }}>
                Tone
              </label>
              <select className="select-field" value={tone} onChange={e => setTone(e.target.value)}>
                {TONES.map(t => <option key={t} value={t}>{t}</option>)}
              </select>
            </div>
            <div>
              <label style={{ display: "block", fontFamily: "'Lato', sans-serif", fontSize: 11, letterSpacing: 2, color: "#c9a84c", textTransform: "uppercase", marginBottom: 8 }}>
                How Many
              </label>
              <div style={{ display: "flex", alignItems: "center", gap: 12 }}>
                <button className="counter-btn" onClick={() => setCount(Math.max(1, count - 1))}>−</button>
                <span style={{ fontFamily: "'Playfair Display', serif", fontSize: 22, color: "#c9a84c", minWidth: 20, textAlign: "center" }}>{count}</span>
                <button className="counter-btn" onClick={() => setCount(Math.min(5, count + 1))}>+</button>
              </div>
            </div>
          </div>

          {/* Error */}
          {error && (
            <div style={{ background: "#3a1520", border: "1px solid #7a2030", borderRadius: 8, padding: "10px 14px", fontFamily: "'Lato', sans-serif", fontSize: 13, color: "#f08090", marginBottom: 16 }}>
              {error}
            </div>
          )}

          <button className="generate-btn" onClick={generate} disabled={loading}>
            {loading ? "Channeling the ritual..." : "✦ Generate Captions ✦"}
          </button>
        </div>

        {/* Loading */}
        {loading && (
          <div style={{ textAlign: "center", padding: 32 }}>
            <div className="spinner" />
            <p style={{ fontFamily: "'Lato', sans-serif", fontSize: 13, color: "#7a6f8a", marginTop: 16 }}>
              Writing in the voice of Home of Rituals...
            </p>
          </div>
        )}

        {/* Results */}
        {captions.length > 0 && (
          <div>
            <div style={{ display: "flex", alignItems: "center", gap: 12, marginBottom: 20 }}>
              <div style={{ flex: 1, height: 1, background: "#2e2650" }} />
              <span style={{ fontFamily: "'Lato', sans-serif", fontSize: 11, letterSpacing: 3, color: "#c9a84c", textTransform: "uppercase" }}>
                {captions.length} Caption{captions.length > 1 ? "s" : ""} Generated
              </span>
              <div style={{ flex: 1, height: 1, background: "#2e2650" }} />
            </div>

            <div style={{ display: "flex", flexDirection: "column", gap: 16 }}>
              {captions.map((c, i) => (
                <div key={i} className="caption-card" style={{ animationDelay: `${i * 0.1}s` }}>
                  <div style={{ display: "flex", alignItems: "flex-start", justifyContent: "space-between", marginBottom: 14, gap: 12 }}>
                    <div style={{ display: "flex", gap: 8, flexWrap: "wrap" }}>
                      <span className="pill">{PLATFORM_ICONS[platform]} {platform}</span>
                      <span className="pill">{PILLAR_ICONS[pillar]} {pillar}</span>
                      <span className="pill" style={{ color: "#9a8ab0" }}>#{i + 1}</span>
                    </div>
                    <button
                      className={`copy-btn ${copied === i ? "copied" : ""}`}
                      onClick={() => copyCaption(c.caption + (c.hashtags ? "\n\n" + c.hashtags : ""), i)}
                    >
                      {copied === i ? "✓ Copied!" : "Copy"}
                    </button>
                  </div>

                  {c.hook && (
                    <div style={{ fontFamily: "'Playfair Display', serif", fontSize: 15, color: "#c9a84c", marginBottom: 10, fontStyle: "italic", borderLeft: "2px solid #c9a84c33", paddingLeft: 12 }}>
                      "{c.hook}"
                    </div>
                  )}

                  <p style={{ fontFamily: "'Lato', sans-serif", fontSize: 14, lineHeight: 1.8, color: "#d4cce0", margin: "0 0 14px", whiteSpace: "pre-wrap" }}>
                    {c.caption}
                  </p>

                  {c.hashtags && (
                    <div style={{ fontFamily: "'Lato', sans-serif", fontSize: 12, color: "#6a5f7a", borderTop: "1px solid #2e2650", paddingTop: 12 }}>
                      {c.hashtags}
                    </div>
                  )}
                </div>
              ))}
            </div>

            <div style={{ textAlign: "center", marginTop: 24 }}>
              <button
                onClick={generate}
                style={{ background: "transparent", border: "1px solid #3a2f5a", color: "#7a6f8a", borderRadius: 10, padding: "10px 24px", fontFamily: "'Lato', sans-serif", fontSize: 13, cursor: "pointer", transition: "all 0.2s" }}
                onMouseEnter={e => { e.target.style.borderColor = "#c9a84c"; e.target.style.color = "#c9a84c"; }}
                onMouseLeave={e => { e.target.style.borderColor = "#3a2f5a"; e.target.style.color = "#7a6f8a"; }}
              >
                ↺ Regenerate New Variations
              </button>
            </div>
          </div>
        )}

        {/* Footer */}
        <div style={{ textAlign: "center", marginTop: 48, fontFamily: "'Lato', sans-serif", fontSize: 11, color: "#3a3250", letterSpacing: 2 }}>
          ✦ HOME OF RITUALS ✦
        </div>
      </div>
    </div>
  );
}
